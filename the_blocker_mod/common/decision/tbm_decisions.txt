	owned_planets_only = yes
	icon = "gfx/interface/icons/decisions/decision_arcology_project.dds"  # Replace this with the art that shows the mag-lev
	enactment_time = 1800
	resources = {
		cost = {
			trigger = {
				NOT = {
					owner = { has_technology = tech_housing_2 }
				}
				engineering_research = value:highway_engineers_to_change_a_lightbulb
				alloys = 2000
				minerals = 5000
			}
		}
		cost = {
			alloys = 2000
			minerals = 5000
		}
	}
	potential = {
		has_deposit = d_collapsed_superhighway
	}

	effect = {
		remove_deposit = d_collapsed_superhighway
	}

	ai_weight = {
		weight = 0
		modifier = {
			add = 1000
			owner = { has_technology = tech_housing_2 }
		}
	}
	
}

					owner = {

							type = food
							amount < 500
						}
					}

				}
			}
		}
	}
}


### Reactor Decisions
#	Open a new super-reactor on your planet and reap the benefits. Safer with better tech!
decision_open_fusion_reactor = {
	owned_planets_only = yes
	icon = "gfx/interface/icons/decisions/decision_drilling_for_gas"

	enactment_time = 720

	resources = {
		cost = {
			alloys = 800
		}
	}

	potential = {
		owner = { has_technology = tech_fusion_power }
		NOT = { has_deposit = d_fusion_reactor	# planet - we don't want two, partly balance partly realism }
	}

	effect = {
		add_deposit = d_fusion_reactor
	}

	ai_weight = {
		weight = 0
		modifier = {
			OR = {
				
				# AI needs the energy
				AND = {
					num_free_districts = {
						type = district_generator
						value < 1
					}
					owner = {
						owner = {
							resource_income_compare = {
								resource = energy
								value < 20
							}
						}
					}
				}

				# Or can benefit from technician boost
				colony_type = col_generator
			}
			
			add = 100
		}
	}
}
