# 51. Poisoned By Kelp (Toxic Kelp)
planet_event = {
	id = tbm.51

	title = tbm.51.name
	desc = tbm.51.desc
	picture = GFX_evt_underwater_ruins
	show_sound = event_alien_nature

	is_triggered_only = yes

	pre_triggers = {
		is_homeworld = no
		has_owner = yes
	}

	trigger = {
		has_deposit = d_toxic_kelp
		any_owned_pop = {
			has_job = colonist
		}
		colony_age > 2	# Try to avoid to kelp-based colony wipe-outs. But not trying *that* hard.
		NOT = { has_planet_flag = toxic_kelp_encounter }
	}

	immediate = {
		random_owned_pop = {
			limit = { has_job = colonist }
			kill_pop = yes
		}
		# Kill a second colonist if aquatic. Potential colony abandonment WAD -- Kelp Is Dangerous!!!
		if = {
			limit = {
				owner.species = { has_trait = trait_aquatic }
			}
			random_owned_pop = {
				limit = { has_job = colonist }
				kill_pop = yes
			}
		}
		hidden_effect = {
			set_timed_planet_flag = {
				flag = toxic_kelp_encounter
				years = 5
			}
		}
	}

	option = {
		name = tbm.51.kelp
	}
}

# 53. Fish Attack (Submerged Ore Veins, possibly others)
planet_event = {
	id = tbm.53

	picture = GFX_evt_unspeakable_horror
	show_sound = dragon_attack_mouth

	hide_window = yes
	is_triggered_only = yes

	pre_triggers = {
		has_owner = yes
		is_homeworld = no
	}

	trigger = {
		has_deposit = d_submerged_ore_veins
		has_district = district_mining
	}

	# Handler
	immediate = {

		this = {
			save_event_target_as = fish_planet
		}

		random_list = {
			15 = {
				modifier = {
					factor = 2
					is_wet = yes
				}
				modifier = {
					factor = 2
					owner_main_species = {
						has_trait = trait_aquatic
					}
				}
				modifier = {
					factor = 4
					is_planet_class = pc_ocean
				}
				modifier = {
					factor = 10
					OR = {
						#is_planet_class = pc_pdaquatic	# Ignore error, PD compatibility bonus.
						has_planet_flag = planet_drenched	# Something unexpected got in there.
						check_variable = {
							which = ice_dumping_happened
							value > 0
						}
					}
				}
				modifier = {
					factor = 0.25
					has_planet_flag = docile_fish
				}

				# Set whether crew will escape or not.
				random_list = {
					50 = {
						modifier = {
							factor = 2
							owner_main_species = { has_trait = trait_aquatic	}
						}
						set_planet_flag = crew_escaped
					}
					50 = {}
				}

				# Fish Attack!
				planet_event = { id = tbm.54 days = 1 }
			}
			85 = {}
		}
	}
}

# 54. Fish Attack! (Message)
planet_event = {
	id = tbm.54

	title = tbm.54.title
	desc = {
		exclusive_trigger = {
			has_planet_flag = crew_escaped
		}
		text = tbm.54.escapes
	}
	desc = {
		exclusive_trigger = {
			NOT = { has_planet_flag = crew_escaped }
		}
		text = tbm.54.eaten
	}
	picture = GFX_evt_unspeakable_horror
	show_sound = dragon_attack_mouth

	is_triggered_only = yes

	option = {
		name = tbm.54.name
		custom_tooltip = tbm.54.worse

		# Figure out how much was in the sub (1 month worth of minerals from planet)
		hidden_effect = {
			every_owned_pop = {
				limit = {
					has_job = miner
				}
				export_trigger_value_to_variable = {
					trigger = pop_produces_resource
					parameters = {
						type = minerals
					}
					variable = @miner_monthly
				}
				change_variable = {		# increment
					which = @miner_output_add
					value = @miner_monthly
				}
			}
		}

		# set_variable = {
		# 	which = @miner_output_mult
		# 	value = prev.miner_output_add
		# }

		# Lose the shipment			
		owner = {
			add_resource = {
				minerals = @miner_output_add
			}	
		}	

		# Miners on the sub die
		if = {
			limit = { 
				NOT = { has_planet_flag = crew_escaped }
			}
			random_owned_pop = {
				limit = {
					has_job = miner
				}
				kill_pop = yes
				kill_pop = yes
			}
		}

		# They say there's worse down there...
		hidden_effect = {
			random_list = {
			
				# Nothing else... for now.
				60 = {
					# And 50% that the fish aren't so bad, really.
					random_list = {
						50 = {
							set_planet_flag = docile_fish
						}
						50 = {}
					}				
				}
	
				# Fish Attacks Continue (Xenofish Attracted to Zro)
				20 = {
					planet_event = { 
						id = tbm.55
						days = 3600
						random = 720 
					}
				}
	
				# There's Always A Bigger Fish
				10 = {
					planet_event = { 
						id = tbm.56 
						days = 3600 
						random = 720 
					}
				}
	
				# Rumours of a Great Monster
				10 = {
					modifier = {
						factor = 0		# Until I finish writing it
						always = yes
					}
					planet_event = { 
						id = tbm.57 
						days = 3600 
						random = 720 
					}
				}
			}
		}
	}
}

### 86. Struck The Motherlode (Handler) (on_district_complete)
planet_event = {
	id = tbm.86

	is_triggered_only = yes
	hide_window = yes

	pre_triggers = {
		has_owner = yes
		is_occupied_flag = no
	}

	trigger = {
		last_district_changed = district_mining
		OR = {
			has_mineral_seam = yes
		}
		is_artificial = no
	}

	immediate = {
		# Get @num_miners
		export_trigger_value_to_variable = {
			trigger = num_assigned_jobs
			parameters = {
				job = miner
			}
			variable = @miners
		}

		# Get deposit with the seam
		random_deposit = {
			limit = {
				has_mineral_seam = yes
			}
			save_event_target_as = mineral_seam
		}

		random_list = {			
			# Struck the Motherlode
			1 = {
				modifier = {
					factor = 0
					OR = {
						owner = { has_country_flag = struck_the_motherlode }	#Can only strike so many motherlodes... and we don't want every planet in the galaxy to get High-Quality Minerals eventually.
						has_planet_flag = motherlode
						has_planet_modifier = pm_mineral_poor
						has_planet_modifier = pm_mineral_rich
						has_planet_modifier = pm_ultra_rich
						has_modifier = mineral_rich # See above
						has_modifier = ultra_rich
					}
				}
				modifier = {
					factor = 2
					OR = {
						has_planet_modifier = pm_asteroid_impacts
						has_planet_modifier = pm_carbon_world
						has_planet_modifier = pm_high_gravity
					}
					
				}
				modifier = {
					add = @miners
				}
				planet_event = {
					id = tbm.87
					days = 60
					random = 30
				}
				set_planet_flag = motherlode
			}
			# Failed Prospects
			9 = {
				modifier = {
					add = @miners
				}
				planet_event = {
					id = tbm.88
					days = 60
					random = 30
				}
			}
			# Standard mining district, normal output
			90 = {}
		}
	}

### 90. Mine Shaft Collapse (Various caves)
planet_event = {
	id = tbm.90

	is_triggered_only = yes

	title = tbm.90.name
	desc = {
		text = tbm.90_desc_robotic_deaths
		trigger = {
			check_variable = {
				which = @casualty_chance
				value > 50
			}
			owner = {
				OR = {
					has_policy_flag = ai_servitude
					is_gestalt = yes
				}
			}
			count_owned_pops = {
				has_job = miner
				is_robotic_species = yes
				count >= 2
			}
		}
	}
	desc = {
		text = tbm.90_desc_miner_deaths
		trigger = {
			check_variable = {
				which = @casualty_chance
				value > 50
			}
		}
	}
	desc = {
		text = tbm.90_desc_no_deaths
		trigger = {
			check_variable = {
				which = @casualty_chance
				value <= 50
			}
		}
	}
	
	immediate = {

		### Damage
		event_target:collapsed_mine = {
			remove_deposit = yes
		}
		
		### Casualties
		if = {
			limit = {
				check_variable = {
					which = @casualty_chance
					value > 50
				}
			}
			if = {
				limit = {
					owner = {
						OR = {
							has_policy_flag = ai_servitude
							is_gestalt = yes
						}
					}
					count_owned_pops = {
						has_job = miner
						is_robotic_species = yes
						count >= 2
					}
				}
				random_owned_pop = {
					limit = {
						has_job = miner
						OR = {
							AND = {
								is_robotic_species = yes
								owner = { is_gestalt = no }
							}
							owner = { is_gestalt = yes }
						}
					}
					kill_pop = yes
					kill_pop = yes
				}
			}
			else = {
				random_owned_pop = {
					limit = {
						has_job = miner
					}
					kill_pop = yes
					kill_pop = yes
				}

				add_modifier = {
					modifier = miners_strike
					days = 720
				}
			}
		}
		else = {
		}
	}
}
