namespace = tbm

### 101. A Shiny Rock (Gatekeeper) (Minerals Fields)
# !!!! IMPORTANT: on_colonized
planet_event = {
	id = tbm.101

	hide_window = yes
	is_triggered_only = yes

	pre_triggers = {
		has_owner = yes
		has_ground_combat = no
		is_homeworld = no
		is_occupied_flag = no
	}

	trigger = {
		has_deposit = d_mineral_fields
		any_owned_pop = {
			is_robot_pop = no
		}
		owner = { is_gestalt = no }
		NOT = { 
			owner = {has_country_flag = a_shiny_rock }
		}
		owner = {
			OR = {
				has_federation = yes
				has_dyson_sphere = yes
				any_owned_planet = {
					is_planet_class = pc_city
				}
				any_owned_starbase = {
					OR = {
						has_starbase_size = starbase_starhold
						has_starbase_size = starbase_starfortress
					}
				}
			}
		}
	}

	immediate = {
		
		random_owned_pop = {
			limit = {
				has_job = colonist
			}
			save_event_target_as = colonizer@species
		}

		# Create lithoid species (check paramaters)
		create_species = {
			name = random
			class = LITHOID
			portrait = lith2
			traits = random
			sapient = yes
			can_be_modified = yes
			namelist = random_class
			effect = {
				save_global_event_target_as = goldem_species
				create_leader = {
					name = random
					class = scientist
					species = this
					set_age = 24
					# Note: Lithy IS eligible to be a normal leader later; we'll get rid of him if not.
					#traits
				}
				last_created_leader = {
					save_event_target_as = Lithy
					exile_leader_as = Lithy
				}
			}
		}

		# Creating a bunch of spoof leaders
		owner = {

			# Rival
			create_leader = {
				name = random
				class = scientist
				species = event_target:colonizer
			}
			last_created_leader = {
				save_event_target_as = Rival
				exile_leader_as = Rival
			}

			# Protagonist
			create_leader = {
				name = random
				class = scientist
				species = event_target:colonizer
				set_age = 11
				#traits
			}
			last_created_leader = {
				save_event_target_as = Character
				exile_leader_as = Character
			}
		}
		
		# Now creating some setting
		planet = {		# Yeah, this is rendundant but you'd be surprised how long it was trying to pull this deposit from owner
			# Local Lithoids
			random_deposit = {
				limit = {
					OR = {
						is_deposit_type = d_mineral_fields
						is_deposit_type = d_basaltic_outcrops
					}
				}
				save_event_target_as = lithoid_deposit
			}
		}
		
		owner = {
			
			# Not-So-Imaginary Structure
			random_list = {
				
				# Ecumenopolis
				25 = {
					modifier = {
						factor = 0
						NOT = {
							owner = {
								any_owned_planet = {
									is_planet_class = pc_city
								}
							}
						}
					}
					random_owned_planet = {
						limit = {
							is_planet_class = pc_city
						}
						save_event_target_as = big_city
					}
				}

				# Federation (must not be leader of it, that would be boring)
				25 = {
					modifier = {
						factor = 0
						NOT = {
							owner = {
								OR = {
									NOT = { exists = federation }
									is_federation_leader = yes
								}
							}
						}
					}
					owner.federation = {
						federation_leader = {
							capital_scope = {
								save_event_target_as = seat_of_power		
							}
						}
					}
				}

				# Dyson Sphere
				25 = {
					modifier = {
						factor = 0
						NOT = {
							has_dyson_sphere = no
						}
					}					
					random_system_megastructure = {
						limit = {
							has_dyson_sphere = yes
						}
						save_event_target_as = sphere@system
					}
				}
				
				#Fallback = Starbase
				25 = {
					modifier = {
						factor = 0
						owner = {
							OR = {
								any_owned_megastructure = {
									has_dyson_sphere = yes
								}
								AND = {
									exists = federation
									NOT = { is_federation_leader = yes }
								}
								any_owned_planet = {
									is_planet_class = pc_city
								}
							}
						}
					}
					owner = {
						random_owned_starbase = {
							limit = {
								NOT = { 
									OR = {
										has_starbase_size = starbase_outpost
										has_starbase_size = starbase_citadel
									}
								}
							}
							weights = {
								base = 10
								modifier = {
									factor = 5
									has_starbase_size = starbase_starfortress
								}
							}
							save_event_target_as = fortress
						}
					}
				}
			}
		}

		this = {
			planet_event = {
				id = tbm.102
			}
		}
	}
}

### 102. A Shiny Rock
planet_event = {
	id = tbm.102

	title = tbm.102.name
	desc = tbm.102.desc
	picture = GFX_evt_crystal_mining
	show_sound = event_machinery

	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			owner = {
				set_country_flag = a_shiny_rock
			}
		}
	}

	option = {
		name = tbm.102.search
		hidden_effect = {
			planet_event = {
				id = tbm.103
				days = 60
			}
		}
	}

	option = {
		name = tbm.102.magic
		custom_tooltip = tbm.102.tooltip.main
		random_list = {
			50 = {
				custom_tooltip = tbm.102.tooltip.magic
				owner = {
					add_resource = {
						unity = 500
					}
				}
			}
			50 = {
				custom_tooltip = tbm.102.tooltip.science
				owner = {
					add_resource = {
						physics_research = 1000
						society_research = 1000
						engineering_research = 1000
					}
				}
			}
		}
	}

	option = {
		name = tbm.102.travel
		allow = {
			hidden_trigger = {
				owner = {
					OR = {
						exists = event_target:ecumenopolis
						exists = event_target:seat_of_power
						exists = event_target:sphere
						exists = event_target:fortress
					}
				}
			}
		}

		event_target:structure = {
			enable_special_project = {
				name = "MEGASTRUCTURE_TOUR"
			}
		}
	}
}
